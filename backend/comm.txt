create table student(rollnumber int NOT NULL PRIMARY KEY, fname varchar(255) NOT NULL, lname varchar(255), password varchar(255), hostel varchar(255) NOT NULL);

create table driver(driverid int NOT NULL PRIMARY KEY, fname varchar(255) NOT NULL, lname varchar(255), username varchar(255), password varchar(255));

create table record(rollnumber int, time TIMESTAMP, distance int, driverid int, tripid int NOT NULL PRIMARY KEY, FOREIGN KEY (rollnumber) REFERENCES student(rollnumber), FOREIGN KEY (driverid) REFERENCES driver(driverid));

create table authority(authid int NOT NULL PRIMARY KEY, fname varchar(255) NOT NULL, lname varchar(255), username varchar(255), password varchar(255));
